// Basic phone preview
const phonePreview = new WebsitePreview({
    container: document.getElementById('myContainer'),
    url: 'https://example.com',
    previewType: 'phone',
    width: 320,
    height: 640
});

// Desktop preview with custom styling
const desktopPreview = WebsitePreview.createDesktopPreview(
    document.getElementById('desktopContainer'),
    'https://github.com',
    {
        width: 800,
        height: 600,
        openInNewTab: true
    }
);

// Custom preview with callbacks
const customPreview = new WebsitePreview({
    container: document.getElementById('customContainer'),
    url: 'https://codepen.io',
    width: 400,
    height: 500,
    borderRadius: '15px',
    onLoad: (preview) => console.log('Website loaded!'),
    onClick: (url, preview) => window.open(url, '_blank')
});
